pipeline {
    agent any

    environment {
        AWS_REGION = "ap-south-1"
        ECR_REPO = "mrmpg-app"
        IMAGE_TAG = "${BUILD_NUMBER}"
        ECR_URL = "${ACCOUNT_ID}.dkr.ecr.${AWS_REGION}.amazonaws.com"
        KOPS_CLUSTER = "kops.k8s.local"
        KOPS_STATE_STORE = "s3://rvkbucket1999"
        ACCOUNT_ID = "123456789012"   // 🔴 replace
        SONAR_HOST_URL = "http://your-sonarqube-url"

        IMAGE_URI = "${ACCOUNT_ID}.dkr.ecr.${AWS_REGION}.amazonaws.com/${ECR_REPO}:${IMAGE_TAG}"
    }

    stages {

        stage('Checkout Code') {
            steps {
                git branch: 'main',
                    url: 'https://github.com/your-repo.git',
                    credentialsId: 'github-creds'
            }
        }

        stage('Code Quality - SonarQube') {
            steps {
                withSonarQubeEnv('SonarQube') {
                    sh """
                    sonar-scanner \
                      -Dsonar.projectKey=mrmpg-app \
                      -Dsonar.sources=. \
                      -Dsonar.host.url=${SONAR_HOST_URL} \
                      -Dsonar.login=${SONAR_TOKEN}
                    """
                }
            }
        }

        stage('Quality Gate') {
            steps {
                timeout(time: 5, unit: 'MINUTES') {
                    waitForQualityGate abortPipeline: true
                }
            }
        }

        stage('Build Docker Image') {
            steps {
                sh "docker build -t ${ECR_REPO}:${IMAGE_TAG} ."
            }
        }

        stage('Login to AWS ECR') {
            steps {
                withCredentials([[
                    $class: 'AmazonWebServicesCredentialsBinding',
                    credentialsId: 'aws-creds'
                ]]) {
                    sh """
                    aws ecr get-login-password --region ${AWS_REGION} \
                    | docker login --username AWS \
                    --password-stdin ${ACCOUNT_ID}.dkr.ecr.${AWS_REGION}.amazonaws.com
                    """
                }
            }
        }

        stage('Tag & Push Image to ECR') {
            steps {
                sh """
                docker tag ${ECR_REPO}:${IMAGE_TAG} ${IMAGE_URI}
                docker push ${IMAGE_URI}
                """
            }
        }

        stage('Helm Package') {
            steps {
                sh """
                helm create mrmpg-chart || true

                # Update values.yaml dynamically
                sed -i 's|repository: .*|repository: ${ACCOUNT_ID}.dkr.ecr.${AWS_REGION}.amazonaws.com/${ECR_REPO}|' mrmpg-chart/values.yaml
                sed -i 's|tag: .*|tag: "${IMAGE_TAG}"|' mrmpg-chart/values.yaml

                helm package mrmpg-chart
                """
            }
        }

        stage('Deploy to Kubernetes') {
            steps {
                sh """
                kubectl apply -f k8s/   # optional if raw manifests exist

                helm upgrade --install mrmpg-release mrmpg-chart \
                  --set image.repository=${ACCOUNT_ID}.dkr.ecr.${AWS_REGION}.amazonaws.com/${ECR_REPO} \
                  --set image.tag=${IMAGE_TAG}
                """
            }
        }
    }

    post {
        always {
            cleanWs()
        }

        success {
            echo "✅ Deployment Successful!"
        }

        failure {
            echo "❌ Pipeline Failed!"
        }
    }
}
